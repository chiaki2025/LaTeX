\item Giải các bất phương trình bậc 2 sau với dấu \textbf{AND} và \textbf{OR}:
\\ \textit{Lưu ý: biểu diễn kết quả trên cả tập nghiệm và trục số}
\begin{enumerate}
    \item \begin{equation*}
        \left\{\begin{aligned}
            &&x^2-2x-24<0\\
            &&12-5x\geq x+9
        \end{aligned}\right.
    \end{equation*}
    \item \begin{equation*}
        \left\{\begin{aligned}
            &&x^2-100>0\\
            &&x^2+8x-105>0
        \end{aligned}\right.
    \end{equation*}
    \item $$25<x^2<64$$
    \item \begin{equation*}
        \left[\begin{array}{l}
        -3x-5\leq10\\
        2x+8\leq16\\
        \end{array}\right.
    \end{equation*}
    \item \begin{equation*}
        \left[\begin{array}{l}
        \frac{2}{3}x-4\leq3\\
        \frac{1}{4}\left(x+8\right)\geq-1\\
        \end{array}\right.
    \end{equation*}
\end{enumerate}
\item Giải các bất phương trình chứa dấu giá trị tuyệt đối sau:

\begin{enumerate}
    \item $$|2x-5|>11$$
    \item $$|7-3x|>2$$
    \item $$\left|\frac{1}{7}-\frac{3}{14}x\right|>\frac{1}{2}$$
    \item $$|3x+4|>-4$$
    \item $$|5x-3|\leq0$$
    \item $$|10x+5|<25$$
    \item $$5|4-3x|-10\leq10$$
\end{enumerate}
